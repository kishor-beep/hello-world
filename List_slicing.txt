List = 'python'
>>> List[-1]   # first character from the end
'n' 
>>> List[-2]   # second character from the end
'o' 
>>> List[1:3]
'yt'
>>> List[2:5]
'tho' 
>>> List[:5]   # slice from the beginning
'pytho' 
>>> List[2:]   # slice until the end
'thon'
>>> List[-3]
'h' 
>>> List[-3:]   # negative index can also be used in slicing
'hon'
