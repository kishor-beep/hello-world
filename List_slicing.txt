w = 'python'
>>> w[-1]   # first character from the end
'n' 
>>> w[-2]   # second character from the end
'o' 
>>> w[1:3]
'yt'
>>> w[2:5]
'tho' 
>>> w[:5]   # slice from the beginning
'pytho' 
>>> w[2:]   # slice until the end
'thon'
>>> w[-3]
'h' 
>>> w[-3:]   # negative index can also be used in slicing
'hon'
